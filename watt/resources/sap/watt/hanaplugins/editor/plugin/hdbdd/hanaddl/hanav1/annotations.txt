@Scope: #ENTITY
annotation Catalog { 
	tableType: String(20) enum {ROW= 'ROW'; COLUMN= 'COLUMN'; GLOBAL_TEMPORARY= 'GLOBAL_TEMPORARY';} default #COLUMN;
	index : array of {
		name : String(128);
		unique : boolean default true;
		order : String(4) enum {ASC='ASC'; DESC='DESC';} default #ASC;
		elementNames : array of String(128);
	};
};

@Scope: #VIEW
annotation ColumnView {
	useCalcEngine : boolean default true;
};
